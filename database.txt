CREATE TABLE Admin (
    ad_id VARCHAR(30) NOT NULL PRIMARY KEY,
    ad_Names VARCHAR(50) NOT NULL,
    ad_Surname VARCHAR(30) NOT NULL,
    ad_Phone VARCHAR(20) NOT NULL,
    ad_Email VARCHAR(50) NOT NULL UNIQUE,
    ad_Church_Key VARCHAR(255) NOT NULL,
    ad_Position VARCHAR(30) NOT NULL,
    ad_Gender VARCHAR(30) NOT NULL,
    ad_Password VARCHAR(255) NOT NULL,
    ad_last_visit VARCHAR(30) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    ad_picture VARCHAR(255) DEFAULT '../images/avatar.png',
    _By VARCHAR(30) NOT NULL,
    _On VARCHAR(30) NOT NULL DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE ch_Member (
    m_id VARCHAR(30) NOT NULL PRIMARY KEY,
    m_Names VARCHAR(50) NOT NULL,
    m_Surname VARCHAR(30) NOT NULL,
    m_Phone VARCHAR(30) NOT NULL,
    m_Email VARCHAR(30) NOT NULL UNIQUE DEFAULT 'NONE',
    m_dateOfBirth VARCHAR(30) NOT NULL,
    m_Status VARCHAR(30) NOT NULL DEFAULT 'ACTIVE',
    m_Position VARCHAR(30) NOT NULL,
    m_Location VARCHAR(30) NOT NULL,
    m_Gender VARCHAR(30) NOT NULL,
    m_About VARCHAR(255) NOT NULL,
    m_Member_Start_Date VARCHAR(30) NOT NULL,
    _By VARCHAR(30) NOT NULL,
    _On VARCHAR(30) NOT NULL DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE activity (
    a_id INT(50) PRIMARY KEY AUTO_INCREMENT,
    activity VARCHAR(255) NOT NULL,
    m_id VARCHAR(50) NOT NULL,
    _By VARCHAR(30) NOT NULL,
    _On VARCHAR(50) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (m_id)
    REFERENCES ch_Members(m_id)
);
CREATE TABLE ch_key (
    k_id INT(30) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    c_key VARCHAR(255) NOT NULL,
    _By VARCHAR(30) NOT NULL,
    _On VARCHAR(50) NOT NULL DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE announcement (
    an_id INT(30) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    an_Announcement VARCHAR(255) NOT NULL,
    _By VARCHAR(30) NOT NULL,
    _On VARCHAR(50) NOT NULL DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE pledge (
    p_id INT(30) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    p_name VARCHAR(255) NOT NULL,
    _By VARCHAR(30) NOT NULL,
    _On VARCHAR(50) NOT NULL DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE member_pledge (
    mp_id INT(30) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    mp_name VARCHAR(255) NOT NULL,
    mp_amount INT(255) NOT NULL,
    mp_item VARCHAR(255) NOT NULL,
    m_id VARCHAR(255) NOT NULL,
    _By VARCHAR(30) NOT NULL,
    _On VARCHAR(50) NOT NULL DEFAULT CURRENT_TIMESTAMP
    FOREIGN KEY (m_id)
    REFERENCES ch_Members(m_id)
);
CREATE TABLE message (
    ms_id INT(30) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    ms_message VARCHAR(255) NOT NULL,
    ms_type VARCHAR(255) NOT NULL,
    ms_seen VARCHAR(255) NOT NULL,
    ms_sender_phone VARCHAR(255) NOT NULL,
    ms_sender_email VARCHAR(255) NOT NULL,
    _By VARCHAR(30) NOT NULL,
    _On VARCHAR(50) NOT NULL DEFAULT CURRENT_TIMESTAMP
);
